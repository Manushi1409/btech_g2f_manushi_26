#include <stdio.h>

int main() {
    // A 3x3 square matrix example
    int matrix[3][3] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };
    int size = 3; // Number of rows/columns

    printf("Principal Diagonal Elements:\n");
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            if (i == j) {
                printf("%d ", matrix[i][j]);
            }
        }
    }
    printf("\n");

    // A more efficient way to print the principal diagonal
    printf("Principal Diagonal (efficient):\n");
    for (int i = 0; i < size; i++) {
        printf("%d ", matrix[i][i]);
    }
    printf("\n");

    return 0;
}
