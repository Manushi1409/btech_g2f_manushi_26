#include <stdio.h>

int main() {
    int r, c, i, j;
    printf("Enter rows and columns: ");
    scanf("%d %d", &r, &c);
    int mat[r][c], trans[c][r];

    // Input elements
    for (i = 0; i < r; ++i)
        for (j = 0; j < c; ++j) scanf("%d", &mat[i][j]);

    // Perform transpose logic: trans[j][i] = mat[i][j]
    for (i = 0; i < r; ++i)
        for (j = 0; j < c; ++j) trans[j][i] = mat[i][j];

    // Print transposed matrix
    printf("\nTranspose:\n");
    for (i = 0; i < c; ++i)
        for (j = 0; j < r; ++j) printf("%d%s", trans[i][j], j == r - 1 ? "\n" : " ");
    return 0;
}
